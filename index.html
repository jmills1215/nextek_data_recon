                    <p>Simple list format - best for data exports and basic analysis</p>

                        <li>Fastest to create and run</li>

                        <li>Ideal for: Data exports, simple lists, record counts</li>

                    <p>Grouped data with subtotals - most versatile for analysis</p>

                    <ul>

                        <li>Up to 3 levels of grouping</li>

                        <li>Subtotals and grand totals</li>

                        <li>Chart compatibility</li>

                        <li>Ideal for: Duplicate detection, trend analysis, dashboards</li>

                    </ul>

                    <ul>

                        <li>Two-dimensional grouping</li>

                        <li>Complex data relationships</li>

                        <li>Advanced summarization</li>

                    </ul>

                <h3>Data Verification with SOQL</h3>

                <div class="alert alert-info">

                    <strong>Best Practice:</strong> Always verify your report results using SOQL queries in Developer Console or Workbench to ensure data accuracy before proceeding with analysis.

                </div>


                <div class="python-example">

<span class="comment"># Essential SOQL verification queries</span>


<span class="string">Verify Total Record Count:</span>

SELECT COUNT() FROM Account WHERE IsPersonAccount = TRUE


<span class="string">Check Recent Data:</span>  

SELECT Id, Name, CreatedDate FROM Account 

WHERE CreatedDate = TODAY AND IsPersonAccount = TRUE


<span class="string">Validate Field Population:</span>

SELECT COUNT() FROM Contact 

WHERE Email != null AND Phone != null


<span class="string">Cross-Reference Report Results:</span>

SELECT Email, COUNT(Id) DupeCount FROM Contact 

GROUP BY Email HAVING COUNT(Id) > 1

                </div>

                <h3>Filtering Techniques</h3>

                <div class="python-example">

<span class="comment"># Common filter patterns for FMO/IMO data</span>


<span class="string">Date Filters:</span>

Created Date equals THIS_MONTH

Last Modified Date equals LAST_N_DAYS:7

License Expiration Date equals NEXT_30_DAYS


<span class="string">Text Filters:</span>

Email contains "@gmail.com"

Phone does not contain "555-0199"

Last Name does not equal ""


<span class="string">Picklist Filters:</span>

Record Type equals "Producer"

Status not equal to "Inactive"

State/Province equals "CA;TX;FL"


<span class="string">Boolean Filters:</span>

Email Opt Out equals FALSE

Person Account equals TRUE

Do Not Call equals FALSE

                <h3>Report Performance Optimization</h3>

                    <strong>Performance Tips:</strong>

                    <ul>

                        <li>Use selective filters to reduce record count</li>

                        <li>Limit to essential fields only</li>

                        <li>Avoid "not equal" filters when possible</li>

                        <li>Use indexed fields in filters (ID, Email, Phone)</li>

                        <li>Consider date ranges for large data sets</li>

                        <li>Schedule large reports to run off-hours</li>

                    </ul>

        function showTab(event, tabName) {

            // Prevent default behavior

            if (event) {

                event.preventDefault();

                event.stopPropagation();

            }

            

            // Hide all tab contents

            const contents = document.getElementsByClassName('tab-content');

            for (let i = 0; i < contents.length; i++) {

                contents[i].classList.remove('active');

            }

            

            // Remove active class from all tabs

            const tabs = document.getElementsByClassName('tab');

            for (let i = 0; i < tabs.length; i++) {

                tabs[i].classList.remove('active');

            }

            

            // Show selected tab content

            const selectedContent = document.getElementById(tabName);

            if (selectedContent) {

                selectedContent.classList.add('active');

            }

            

            // Add active class to clicked tab

            const clickedTab = event ? event.target : event;

            if (clickedTab && clickedTab.classList) {

                clickedTab.classList.add('active');

            } else {

                // Fallback: find the tab by content

                for (let i = 0; i < tabs.length; i++) {

                    if (tabs[i].onclick && tabs[i].onclick.toString().includes(tabName)) {

                        tabs[i].classList.add('active');

                        break;

                    }

                }

            }

            

            return false;

        }

        function initializeTabs() {

                    // Get tab name from onclick attribute as fallback

                    if (onclickAttr) {

                        const match = onclickAttr.match(/showTab\(event,\s*'([^']+)'\)/);

                        if (match) {

                            showTab(e, match[1]);

                        }
